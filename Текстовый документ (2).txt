def find_sublist(main_list, sublist):
    """Поиск первого вхождения подмассива"""
    if not isinstance(main_list, (list, tuple)) or not isinstance(sublist, (list, tuple)):
        raise TypeError("Аргументы должны быть списками или кортежами")
    
    if not sublist:  # Пустой подмассив
        return 0
    if not main_list:  # Пустой основной список
        return -1
    
    for i in range(len(main_list) - len(sublist) + 1):
        if main_list[i:i + len(sublist)] == sublist:
            return i
    return -1


def find_all_sublists(main_list, sublist):
    if not isinstance(main_list, (list, tuple)) or not isinstance(sublist, (list, tuple)):
        raise TypeError("Аргументы должны быть списками или кортежами")
    
    positions = []
    if not sublist or not main_list:
        return positions
    
    for i in range(len(main_list) - len(sublist) + 1):
        if main_list[i:i + len(sublist)] == sublist:
            positions.append(i)
    return positions
